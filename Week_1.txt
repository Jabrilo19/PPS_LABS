# 1, 2 Turn on CSV mode and say that header is on in the CSV file and Create a table called bmi_info with the syntax: create table bmi_info("lastName" TEXT, "firstName" TEXT, "height" REAL, "weight" REAL);
sqlite> CREATE TABLE bmi_info("lastname" TEXT, "firstName" TEXT, "height" REAL, "weight" REAL);
sqlite> .mode csv
sqlite> .header on
